/**
 * Minimal scene sequencer for p5 sketches.
 *
 * A scene is any object with optional `setup()`, `update(dt)`, `draw()`, and
 * `resize()` methods. `SceneQueueObject` lets sketches keep each topic in its
 * own object, then move through those scenes with `next()` and `previous()`.
 *
 * @example
 * const scenes = SceneQueue();
 * scenes.push(createDihedralScene());
 * scenes.push(createSymmetricScene());
 * scenes.run();
 *
 * function draw() {
 *   background(5, 16, 32);
 *   scenes.update();
 *   scenes.draw();
 * }
 */
class SceneQueueObject {
  constructor() {
    this.steps = [];
    this.stepIndex = 0;
    this.currentIndex = -1;
    this.current = null;
    this.elapsed = 0;
    this.running = false;
    this.loopSteps = false;
  }

  /**
   * Adds a scene to the queue.
   *
   * @param {Object|string} scene - Scene object, or a string command.
   * @param {Object} [options] - Scene options.
   * @param {number|null} [options.duration=null] - Optional auto-advance time in seconds.
   * @returns {SceneQueueObject} This queue.
   */
  push(scene, options = {}) {
    if (typeof scene === "string") {
      this.steps.push(Object.assign({ type: scene }, options));
      return this;
    }

    this.steps.push({
      type: "scene",
      scene,
      duration: options.duration === undefined ? null : options.duration
    });
    return this;
  }

  /**
   * Starts the scene queue at the first scene.
   *
   * @param {Object} [options] - Run options.
   * @param {boolean} [options.loop=false] - Wrap at the ends when moving between scenes.
   * @returns {SceneQueueObject} This queue.
   */
  run(options = {}) {
    this.running = true;
    this.loopSteps = Boolean(options.loop);
    this.stepIndex = 0;
    this.currentIndex = -1;
    this.current = null;
    this.elapsed = 0;
    this.goTo(0);
    return this;
  }

  /**
   * Updates the current scene and handles optional timed scene changes.
   *
   * @param {number} [dt] - Delta time in seconds.
   * @returns {SceneQueueObject} This queue.
   */
  update(dt) {
    const seconds = dt || deltaTime / 1000 || 1 / 60;

    if (this.current && typeof this.current.update === "function") {
      this.current.update(seconds);
    }

    if (!this.running || !this.current) return this;

    this.elapsed += seconds;
    if (this.currentDuration !== null && this.elapsed >= this.currentDuration) {
      this.next();
    }

    return this;
  }

  /**
   * Draws the current scene.
   *
   * @returns {SceneQueueObject} This queue.
   */
  draw() {
    if (this.current && typeof this.current.draw === "function") {
      this.current.draw();
    }
    return this;
  }

  /**
   * Forwards resize events to the current scene.
   *
   * @returns {SceneQueueObject} This queue.
   */
  resize() {
    if (this.current && typeof this.current.resize === "function") {
      this.current.resize();
    }
    return this;
  }

  startNextStep() {
    return this.goTo(this.stepIndex);
  }

  /**
   * Moves to a specific scene index.
   *
   * @param {number} index - Zero-based scene index.
   * @returns {SceneQueueObject} This queue.
   */
  goTo(index) {
    if (this.steps.length === 0) {
      this.running = false;
      return this;
    }

    if (this.loopSteps) {
      index = (index + this.steps.length) % this.steps.length;
    } else {
      index = Math.max(0, Math.min(index, this.steps.length - 1));
    }

    if (index === this.currentIndex && this.current) {
      return this;
    }

    const step = this.steps[index];
    this.currentIndex = index;
    this.stepIndex = index + 1;

    if (step.type === "scene") {
      this.current = step.scene;
      this.currentDuration = step.duration;
      this.elapsed = 0;
      if (this.current && typeof this.current.setup === "function") {
        this.current.setup();
      }
      return this;
    }

    return this.goTo(index + 1);
  }

  /**
   * Moves to the next scene.
   *
   * @returns {SceneQueueObject} This queue.
   */
  next() {
    return this.goTo(this.currentIndex + 1);
  }

  /**
   * Moves to the previous scene.
   *
   * @returns {SceneQueueObject} This queue.
   */
  previous() {
    return this.goTo(this.currentIndex - 1);
  }
}

/**
 * Creates a scene queue.
 *
 * @returns {SceneQueueObject}
 */
function SceneQueue() {
  return new SceneQueueObject();
}
